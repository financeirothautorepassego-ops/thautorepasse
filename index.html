function repComissoes(){
  var meses=getAllMonths().reverse();
  var mesFiltro=repMonth==='todos'?null:repMonth;
  var mF=financeiro.filter(function(f){
    var cat=(f.categoria||'').toLowerCase();
    var isComissao=cat==='comissao'&&f.tipo==='saida';
    if(mesFiltro)return isComissao&&monthKey(f.date)===mesFiltro;
    return isComissao;
  });

  // Agrupar por parceiro (consolidando nomes similares)
  var porParceiro={};
  mF.forEach(function(f){
    var nome=f.parceiroNome||f.descricao||'Sem parceiro';
    if(!porParceiro[nome])porParceiro[nome]={total:0,qtd:0,lancamentos:[]};
    porParceiro[nome].total+=Number(f.valor||0);
    porParceiro[nome].qtd+=1;
    porParceiro[nome].lancamentos.push(f);
  });

  var totalGeral=mF.reduce(function(a,f){return a+Number(f.valor||0);},0);
  var totalParceiros=Object.keys(porParceiro).length;

  // ── HEADER ────────────────────────────────────────────────────────

  if(meses.length&&!mesFiltro){
